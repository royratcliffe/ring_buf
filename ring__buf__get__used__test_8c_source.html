13 * Define and reset ring buffer.
14 * The base index is set to sizeof(int[2]) to test non-zero bases.
15 */
16 RING_BUF_DEFINE_STATIC(buf, sizeof(int[4]));
17 ring_buf_reset(&buf, sizeof(int[2]));
18
19 /*
20 * Put four integers into the buffer. Put them one at a time. Putting "all or
21 * none" automatically acknowledges the put zone. Assert that each put
22 * operation succeeds.
23 */
24 for (int y = 1; y <= 4; y++)
25 {
26 int err = ring_buf_put_all(&buf, &y, sizeof(y));
27 assert(err == 0);
28 }
29
30 /*
31 * Get all the used space in the buffer. Copy the data into a dynamically
32 * allocated array. Assert that the correct amount of data is copied.
33 *
34 * Does not acknowledge the get zone on purpose.
35 */
36 ring_buf_size_t len = ring_buf_used_space(&buf) / sizeof(int);
37 assert(len == 4);
38 int *data = malloc(len * sizeof(int));
39 assert(data != NULL);
40 ring_buf_size_t ack = ring_buf_get(&buf, data, len * sizeof(int));
41 assert(ack == len * sizeof(int));
42 assert(ring_buf_get_ack(&buf, 0) == 0);
43
44 /*
45 * Verify the copied data. The data should be 1, 2, 3, 4 in that order.
46 */
47 assert(data[0] == 1);
48 assert(data[1] == 2);
49 assert(data[2] == 3);
50 assert(data[3] == 4);
51
52 /*
53 * Print the data. The loop variable t is of type ring_buf_size_t to match the
54 * type used in the ring buffer API. It stands for time, as in a time series.
55 * Get each element using ring_buf_get_all to demonstrate its use. It
56 * acknowledges each get operation and empties the buffer.
57 */
58 for (ring_buf_size_t t = 0; t < len; t++)
59 {
60 int y;
61 int err = ring_buf_get_all(&buf, &y, sizeof(y));
62 assert(err == 0);
63 assert(data[t] == y);
64 /*
65 * Format as %zu for size_t type.
66 */
67 (void)printf("data[%zu] = %d\n", t, data[t]);
68 }
69
70 /*

72 */
